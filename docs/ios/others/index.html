自定义事件​

   //引力
    //无参数
    SkyIsland.Analytics.LogGravityEvent("DailyReward");
    //有参数
    Dictionary<string, object> dic = new Dictionary<string, object>
    {
        { "label", "白银刀" },
        { "level", 10 },
        { "day", 3},
        //...
    };
    SkyIsland.Analytics.LogGravityEvent("DailyReward", dic);

用户属性​

    //初始化用户属性
    Gravity.UserAttributeSetOnce(new Dictionary<string, object>() {
        {"user_total_coins_earned", cost}
    });
    //设置用户属性
    Gravity.UserAttributeSet(new Dictionary<string, object>() {
        {"user_total_coins_earned", cost}
    });
    //累加用户属性
    Gravity.UserAttributeAdd(new Dictionary<string, object>() {
        {"user_total_coins_earned", cost}
    });
    //重置用户属性
    Gravity.UserAttributeUnset(new Dictionary<string, object>() {
        {"user_total_coins_earned", cost}
    });
    //清空用户属性
    Gravity.UserAttributeDelete(new Dictionary<string, object>() {
        {"user_total_coins_earned", cost}
    });
    //追加用户属性
    Gravity.UserAttributeAppend(new Dictionary<string, object>() {
        {"user_total_coins_earned", cost}
    });
    //去重追加用户属性
    Gravity.UserAttributeUniqAppend(new Dictionary<string, object>() {
        {"user_total_coins_earned", cost}
    });
    //用户属性取最大值
    Gravity.UserAttributeNumberMax(new Dictionary<string, object>() {
        {"user_total_coins_earned", cost}
    });
    //用户属性取最小值
    Gravity.UserAttributeNumberMin(new Dictionary<string, object>() {
        {"user_total_coins_earned", cost}
    });

反馈按钮

需要在游戏中添加反馈按钮，方便用户反馈问题或提交建议。

将SkyIslandFeedbackButton放置到需要显示反馈按钮的界面上的合适位置。或者将FeedbackButton的Click事件添加到自己定义的反馈按钮上。
